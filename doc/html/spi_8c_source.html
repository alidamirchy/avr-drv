00108 void spi_set_operation_mode(spi_op_mode_t mode)
00109 {
00110     switch (mode)
00111     {
00112     case spi_op_mode_single_master:
00113         SCK_PORT |= (1 << SCK_BIT); // set SCK hi
00114         //MOSI & SCK as output
00115         SCK_DDR |= (1 << SCK_BIT);
00116         MOSI_DDR |= (1 << MOSI_BIT);
00117         SS_DDR |= (1 << SS_BIT); // SS as output, will prevent fallback to slave
00118         SPCR |= (1 << MSTR);
00119         break;
00120     case spi_op_mode_multi_mMaster:
00121         SCK_PORT |= (1 << SCK_BIT); // set SCK hi
00122         //MOSI & SCK as output
00123         SCK_DDR |= (1 << SCK_BIT);
00124         MOSI_DDR |= (1 << MOSI_BIT);
00125         SS_DDR &= ~_BV(SS_BIT); // SS as input
00126         SS_PORT |= _BV(SS_BIT); // Activate pull-up on SS
00127         SPCR |= (1 << MSTR);
00128         break;
00129     case spi_op_mode_slave:
00130         SS_DDR &= ~_BV(SS_BIT); // SS as input
00131         SS_PORT |= _BV(SS_BIT); // Activate pull-up on SS
00132         MISO_DDR |= (1 << MISO_BIT); //set MISO as output
00133         SPCR &= ~(1 << MSTR);
00134         break;
00135     default:
00136         break;
00137     }
