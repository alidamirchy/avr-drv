00054 void twi_set_bitrate(uint32_t clkio, uint8_t uwBitrateKHz)
00055 {
00056 
00057         if((clkio/1000/(16+2*0xFF)) <= uwBitrateKHz)
00058         {
00059                 TWBR = ((clkio/1000/uwBitrateKHz - 16)/(2));
00060                 TWSR &= ~(_BV(TWPS1) | _BV(TWPS0));
00061         }
00062         else if((clkio/1000/(16+2*0xFF*4)) <= uwBitrateKHz)
00063         {
00064                 TWBR = ((clkio/1000/uwBitrateKHz - 16)/(2*4));
00065                 TWSR &= ~(_BV(TWPS1));
00066                 TWSR |= _BV(TWPS0);
00067         }
00068         else if((clkio/1000/(16+2*0xFF*16)) <= uwBitrateKHz)
00069         {
00070                 TWBR = ((clkio/1000/uwBitrateKHz - 16)/(2*16));
00071                 TWSR &= ~(_BV(TWPS0));
00072                 TWSR |= _BV(TWPS1);
00073         }
00074         else if((clkio/1000/(16+2*0xFF*64)) <= uwBitrateKHz)
00075         {
00076                 TWBR = ((clkio/1000/uwBitrateKHz - 16)/(2*64));
00077                 TWSR |= _BV(TWPS0) | _BV(TWPS1);
00078         }
00079         else
00080         {
00081                 asm("break");
00082         }
00083 
00084 }
