00874 int can_set_baud_rate(uint32_t ulBaudrate, uint32_t ulClkFreq,
00875         uint8_t ubSamplingRate, uint8_t ubTsjw)
00876 {
00877     uint8_t ubDivider;
00878     uint8_t ubTbit;
00879     uint8_t ubTprs;
00880     uint8_t ubTphs1;
00881     uint8_t ubTphs2;
00882     uint8_t ubBRP = 1;
00883 
00884     ubDivider = ulClkFreq / ulBaudrate; /* Trouve le diviseur total de la fréquence du Can */
00885 
00886     ubTbit = ubDivider;
00887     while ((ubTbit >= CAN_MAX_TQ) | (ubTbit <= CAN_MIN_TQ))
00888     {
00889         ubTbit = (ubTbit >> 1);
00890         ubBRP *= 2;
00891     }
00892 
00893     ubTphs2 = (ubTbit * (100 - ubSamplingRate) / 100);
00894     ubTphs1 = 0;
00895 
00896     for(ubTprs = 8; ubTprs > 0; ubTprs--)
00897     {
00898         if((1 + ubTprs + ubTphs2 + ubTphs2) <= (ubTbit + 8 - ubTphs2)
00899                         && ((ubTbit - 1 - ubTprs - ubTphs2) >= ubTphs2))
00900         {
00901                 ubTphs1 = ubTbit - 1 - ubTprs - ubTphs2;
00902                 break;
00903         }
00904     }
00905 
00906     if (1 == ubBRP)
00907     {
00908         ubTphs1 = ubTphs2 + 1;
00909         ubTphs2 -= 1;
00910     }
00911 
00912     if (ubDivider == (ubBRP) * (ubTprs + ubTphs1 + ubTphs2 + 1))
00913     {
00914         CANBT1 = ((ubBRP-1) << BRP0);
00915                 CANBT2 = ((ubTprs-1) << PRS0) | ((ubTsjw-1) << SJW0);
00916                 CANBT3 = ((ubTphs2-1) << PHS20) | ((ubTphs1-1) << PHS10) | (1 << SMP);
00917         return 0;
00918     }
00919     else
00920     {
00921         avr_drv_errno = EBAUDRATE;
00922         return -1;
00923     }
